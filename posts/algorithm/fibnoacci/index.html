Fibonacci

递归

uint64_t fibonacci(uint64_t n)
{
    if (n > 1) {
        return fibonacci(n -1) + fibonacci(n -2);
    } else {
        return n;
    }
}

非递归

uint64_t fibonacci_1(uint64_t n)
{
    if (n < 2) { return n; }

    uint64_t fn  = 0;
    uint64_t fn0 = 0;
    uint64_t fn1 = 1;

    uint64_t count = 2;
    while (count <= n) {
        fn  = fn0 + fn1;
        fn0 = fn1;
        fn1 = fn;
        count++;
    }
    return fn;
}

矩阵

std::pair<uint64_t, uint64_t> fibonacci_3(size_t n)
{
    // 返回F_{n}和F_{n + 1}, 注意是对UINT64_MAX取模的结果.
    if (n > 0) {
        auto PF = fibonacci_3(n / 2);   // 基于递归求解.
        auto t0 = PF.first;
        auto t1 = PF.second;
        if (n % 2 == 1) {
            // 奇数情况.
            return {t0 * t0 + t1 * t1,    (2 * t0 + t1) * t1 };
        } else {
            return {(2 * t1 - t0) * t0,   t0 * t0 + t1 * t1 };
        }
    }

    return {0, 1};
}
